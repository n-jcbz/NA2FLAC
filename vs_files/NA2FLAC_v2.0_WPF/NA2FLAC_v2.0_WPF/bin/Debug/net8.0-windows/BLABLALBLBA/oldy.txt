<Style TargetType="TextBox">
    <Setter Property="Background" Value="#FF1E1E1E"/>
    <Setter Property="Foreground" Value="White"/>
    <Setter Property="BorderBrush" Value="#FF555555"/>
    <Setter Property="BorderThickness" Value="1"/>
    <Setter Property="Padding" Value="5"/>
    <Setter Property="Template">
        <Setter.Value>
            <ControlTemplate TargetType="TextBox">
                <Border Background="{TemplateBinding Background}"
                BorderBrush="{TemplateBinding BorderBrush}"
                BorderThickness="{TemplateBinding BorderThickness}"
                CornerRadius="5">
                    <Grid>
                        <!-- real text host -->
                        <ScrollViewer x:Name="PART_ContentHost" />

                        <!-- watermark: collapsed by default, shown by trigger only when Text is empty AND not focused -->
                        <TextBlock x:Name="FolderMark"
                           Text="Enter directory here..."
                           Opacity="0.5"
                           Padding="{TemplateBinding Padding}"
                           HorizontalAlignment="Left"
                           VerticalAlignment="Center"
                           Visibility="Visible"
                           IsHitTestVisible="False"
                           Foreground="{TemplateBinding Foreground}"
                           Margin="5,0,0,0"/>
                    </Grid>
                </Border>

                <ControlTemplate.Triggers>
                    <!-- Show watermark ONLY when Text.Length == 0 AND the TextBox is not focused -->
                    <MultiDataTrigger>
                        <MultiDataTrigger.Conditions>
                            <Condition Binding="{Binding Text.Length, RelativeSource={RelativeSource TemplatedParent}}" Value="0"/>
                            <Condition Binding="{Binding IsKeyboardFocusWithin, RelativeSource={RelativeSource TemplatedParent}}" Value="False"/>
                        </MultiDataTrigger.Conditions>
                        <Setter TargetName="FolderMark" Property="Visibility" Value="Visible"/>
                    </MultiDataTrigger>

                    <!-- optional: hide watermark while mouse is over (so it doesn't overlap caret when focusing with mouse) -->
                    <MultiDataTrigger>
                        <MultiDataTrigger.Conditions>
                            <Condition Binding="{Binding Text.Length, RelativeSource={RelativeSource TemplatedParent}}" Value="0"/>
                            <Condition Binding="{Binding IsMouseOver, RelativeSource={RelativeSource TemplatedParent}}" Value="True"/>
                        </MultiDataTrigger.Conditions>
                        <Setter TargetName="FolderMark" Property="Visibility" Value="Collapsed"/>
                    </MultiDataTrigger>
                </ControlTemplate.Triggers>
            </ControlTemplate>
        </Setter.Value>
    </Setter>
</Style>
